Overview

Handles translating a human-readable place name into a set of coordinates. Data retrieved from the pickpoint.io API - Â© OpenStreetMap contributors#

london = new Geocoder "London, UK"
london.search()
.then (result) ->
  console.log "Longitude: #{result.lon}"
  console.log "Latitude: #{result.lat}"
.catch (error) ->
  console.error "Couldn't get coords: #{error}"
